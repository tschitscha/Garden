    document.getElementById('year').textContent = new Date().getFullYear();

    // Open modal & WhatsApp flow
    const ctaFree = document.getElementById('ctaFree');
    const submitBtn = document.getElementById('submitBtn');
    const modal = document.getElementById('modal');
    const closeModal = document.getElementById('closeModal');
    const openWa = document.getElementById('openWa');
    const leadForm = document.getElementById('leadForm');
    const callBtn = document.getElementById('callBtn');

    function sanitizePhone(input) {
      return input.replace(/[^+\d]/g, '');
    }

    function validatePhone(phone) {
      // Improved validation: must start with + and contain 8-15 digits
      return /^\+\d{8,15}$/.test(phone);
    }

    function prepareWaLink(name, phone, country, notes) {
      const base = "https://wa.me/36307148961";
      const msg = `Здравствуйте! Я заказал бесплатную концепцию сада.%0AИмя: ${encodeURIComponent(name || 'не указано')}%0AТелефон: ${encodeURIComponent(phone)}%0AСтрана: ${encodeURIComponent(country)}%0AПожелания: ${encodeURIComponent(notes || '—')}`;
      return base + "?text=" + msg;
    }

    function openModal() {
      modal.style.display = "flex";
      document.body.style.overflow = 'hidden'; // Prevent scrolling
    }

    function closeModalFn() {
      modal.style.display = "none";
      document.body.style.overflow = 'auto'; // Restore scrolling
    }

    // Scroll to form when CTA button is clicked
    ctaFree.addEventListener('click', () => {
      document.getElementById('name').focus();
      document.querySelector('.hero-right').scrollIntoView({
        behavior: 'smooth',
        block: 'center'
      });
    });

    // Handle form submission
    leadForm.addEventListener('submit', (e) => {
      e.preventDefault();
      
      const name = document.getElementById('name').value.trim();
      let phone = document.getElementById('phone').value.trim();
      const country = document.getElementById('country').value;
      const notes = document.getElementById('notes').value.trim();
      const consent = document.getElementById('consent').checked;

      if (!consent) {
        alert('Пожалуйста, согласитесь на обработку контактов для связи.');
        return;
      }

      if (!phone || !validatePhone(sanitizePhone(phone))) {
        alert('Пожалуйста, укажите корректный телефон в международном формате (например, +36 30 123 4567).');
        document.getElementById('phone').focus();
        return;
      }

      phone = sanitizePhone(phone);

      // Show loading state
      submitBtn.classList.add('loading');
      submitBtn.disabled = true;

      // Simulate API call
      setTimeout(() => {
        // Reset loading state
        submitBtn.classList.remove('loading');
        submitBtn.disabled = false;

        // Show success modal
        openModal();

        // Pre-fill WhatsApp message when user clicks Open WhatsApp
        openWa.onclick = function() {
          const link = prepareWaLink(name, phone, country, notes);
          window.open(link, '_blank');
        };

        // Reset form
        leadForm.reset();

        // Log to console (replace with actual API call in production)
        console.log('LEAD SUBMITTED:', {
          name: name || 'не указано',
          phone,
          country,
          notes: notes || '—',
          timestamp: new Date().toISOString()
        });
      }, 1000);
    });

    // WhatsApp button handler
    callBtn.addEventListener('click', function() {
      window.open('https://wa.me/36307148961', '_blank');
    });

    // Modal handlers
    closeModal.addEventListener('click', closeModalFn);
    modal.addEventListener('click', function(e) {
      if (e.target === modal) closeModalFn();
    });

    // Close modal with Escape key
    document.addEventListener('keydown', function(e) {
      if (e.key === 'Escape') closeModalFn();
    });

    // Smooth scrolling for anchor links
    document.querySelectorAll('a[href^="#"]').forEach(anchor => {
      anchor.addEventListener('click', function (e) {
        e.preventDefault();
        const target = document.querySelector(this.getAttribute('href'));
        if (target) {
          target.scrollIntoView({
            behavior: 'smooth',
            block: 'start'
          });
        }
      });
    });
  </script>
</body>
</html>
