    // Insert current year

    const openWa = document.getElementById('openWa');

    function sanitizePhone(input) {
      return input.replace(/[^+\d]/g,'');
    }

    function validatePhone(phone){
      // basic validation: must have at least 8 digits
      const digits = phone.replace(/\D/g,'');
      return digits.length >= 8;
    }

    function prepareWaLink(name, phone, country, notes){
      const base = "https://wa.me/36307148961";
      const msg = `Здравствуйте! Я заказал бесплатную концепцию сада.%0AИмя: ${encodeURIComponent(name)}%0AТелефон: ${encodeURIComponent(phone)}%0AСтрана: ${encodeURIComponent(country)}%0AПожелания: ${encodeURIComponent(notes)}`;
      return base + "?text=" + msg;
    }

    function openModal(){
      modal.style.display = "flex";
    }
    function closeModalFn(){ modal.style.display = "none"; }

    ctaFree.addEventListener('click', () => {
      document.getElementById('name').focus();

    });

    submitBtn.addEventListener('click', () => {
      const name = document.getElementById('name').value.trim();
      let phone = document.getElementById('phone').value.trim();

      const notes = document.getElementById('notes').value.trim();
      const consent = document.getElementById('consent').checked;

      if(!consent){
        alert('Пожалуйста, согласитесь на обработку контактов для связи.');
        return;
      }
      if(!phone || !validatePhone(phone)){
        alert('Пожалуйста, укажите корректный телефон (минимум 8 цифр).');
        document.getElementById('phone').focus();
        return;
      }
      phone = sanitizePhone(phone);

      // Show modal
      openModal();

      // Pre-fill WhatsApp message when user clicks Open WhatsApp
      openWa.onclick = function(){
        const link = prepareWaLink(name || '—', phone, country, notes || '—');
        window.open(link, '_blank');
      };

    });
